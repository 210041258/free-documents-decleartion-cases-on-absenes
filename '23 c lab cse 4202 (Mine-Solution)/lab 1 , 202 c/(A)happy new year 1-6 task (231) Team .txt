#include<stdio.h>
int start(int n);
int check(int *z);

int main()
{
    int n;
    scanf("%d",&n);
    int x;
    x = start(n);
    printf("%d",x);
    return 0;
}

int start(int n)
{
    int sum = 0;
    int x,y;
    int z[3];
    if((1<=n)&&(n<=1000))
    {
        for(x=0; x<n; x++)
        {
            for(y=0; y<3; y++)
            {
                scanf("%d",&z[y]);
            }
            if(check(z))
            {
                ++sum;
            }
        }
    }
    return sum;
}
int check(int *z)
{
    int y=0,m=0;
    for(y=0; y<3; y++)
    {
        if(z[y]!=0)
        {
            m++;
        }
    }
    if(m>=2)
    {
        return 1;
    }
    return 0;
}
